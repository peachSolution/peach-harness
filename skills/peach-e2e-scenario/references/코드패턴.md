# 코드 패턴

E2E 시나리오 파일의 표준 코드 패턴.
대상 프로젝트의 프레임워크에 맞게 셀렉터와 대기 전략을 조정한다.

## 기본 구조

```javascript
/**
 * {시나리오 설명}
 *
 * 사전조건:
 *   - Chrome Beta CDP 모드 실행 (./e2e.sh chrome)
 *   - 브라우저에서 로그인 완료
 *   - {필요한 페이지} 접속
 *
 * 실행: ./e2e.sh run 시나리오/{카테고리}/{파일명}.js
 */
const { connect, setDialogHandler } = require('../../lib/connect');

(async () => {
  console.log('🚀 [{시나리오명}] 시작');

  let exitCode = 0;

  const { page, origin, defaultDialogHandler } = await connect();
  console.log(`📍 현재 페이지: ${page.url()}`);

  // 페이지 체크 — 안전한 기본값은 자동 이동 금지
  if (!page.url().includes('/expected/path')) {
    console.log('👀 현재 탭이 예상 경로가 아닙니다.');
    console.log(`📍 현재 URL: ${page.url()}`);
    throw new Error('예상 경로가 아님 — 사용자 확인 후 다시 실행하세요.');
    // 비민감 세션 + 사용자 실행 승인 확보 시에만 아래 자동 이동 사용:
    // await page.goto(`${origin}/expected/path`);
    // await page.waitForURL('**/expected/path');
  }

  // native dialog가 필요할 때만 교체 패턴으로 설치
  let dialogHandler = null;
  // 예:
  // dialogHandler = setDialogHandler(page, defaultDialogHandler, async (dialog) => {
  //   console.log(`📢 다이얼로그: ${dialog.message()}`);
  //   try { await dialog.accept(); } catch (_) {}
  // });

  try {
    // ── 1. 첫번째 단계 ────────────────────────────────────
    console.log('\n📋 [1] 단계 설명...');
    // 조작 코드
    console.log('✅ 완료');

    // ── 2. 두번째 단계 ────────────────────────────────────
    console.log('\n📋 [2] 단계 설명...');
    // 조작 코드
    console.log('✅ 완료');

    console.log('\n✨ [{시나리오명}] 완료!');
    console.log('👀 화면을 확인하세요.');

  } catch (e) {
    exitCode = 1;
    console.error('\n❌ 에러:', e.message);
    console.log('💡 현재 페이지 상태를 확인하고 다시 시도하세요.');
  } finally {
    if (dialogHandler) {
      setDialogHandler(page, dialogHandler, defaultDialogHandler);
    }
    process.exit(exitCode);
  }
})();
```

> **dialog handler 설치 규칙**: 단순 조회 시나리오에는 handler를 넣지 않는다.
> dialog가 필요한 경우에만 `setDialogHandler(page, defaultDialogHandler, myHandler)`로 교체하고,
> finally에서 `setDialogHandler(page, myHandler, defaultDialogHandler)`로 반드시 원복한다.

> `require` 경로는 시나리오 파일 위치에 따라 `lib/connect.js`까지의 상대경로로 조정한다.

## 셀렉터 우선순위

1. `getByTestId()` — 가장 안정적 (data-testid 속성)
2. `getByRole()` — 시맨틱 접근 (button, dialog, combobox 등)
3. `getByLabel()` — 폼 필드 레이블
4. `getByPlaceholder()` — placeholder 텍스트
5. `getByText()` — 텍스트 내용
6. `#id` — id 속성
7. `[data-*]` — 데이터 속성

> **주의**: CSS-in-JS 해시 클래스 (`.sc-xxxxx`, Tailwind 동적 클래스) 직접 사용 금지 — 빌드마다 변경됨

## 버튼 클릭 패턴

```javascript
// 일반 버튼 — getByRole 우선
await page.getByRole('button', { name: '저장' }).click();
await page.getByRole('button', { name: '검색' }).click();
await page.getByRole('button', { name: '확인' }).click();

// 링크
await page.getByRole('link', { name: '목록' }).click();

// 텍스트로 찾기 (버튼이 아닌 요소)
await page.getByText('완료').click();
```

## 입력 패턴

```javascript
// 텍스트 입력
await page.getByLabel('이름').fill('홍길동');
await page.getByPlaceholder('검색어 입력').fill('검색어');

// select (combobox role)
await page.getByRole('combobox', { name: '상태' }).selectOption('완료');

// checkbox
await page.getByRole('checkbox', { name: '전체선택' }).check();

// 반응형 이벤트 수동 발생 (필요한 경우)
await page.locator('#field').evaluate(el => {
  el.dispatchEvent(new Event('input', { bubbles: true }));
});
```

## 페이지 이동 대기 패턴

### SPA (Vue/React/Next.js — 클라이언트 라우팅)

```javascript
// URL 변경 대기
await page.getByRole('button', { name: '다음 단계' }).click();
await page.waitForURL('**/step2', { timeout: 5000 });

// 특정 요소 등장 대기
await page.getByRole('heading', { name: '2단계' }).waitFor({ timeout: 5000 });

// networkidle은 초기 진입 시에만 허용
await page.goto(`${origin}/path`);
await page.waitForLoadState('networkidle');
```

### MPA (PHP/Django/Rails — 풀 리로드)

```javascript
await page.locator('a', { hasText: '다음' }).click();
await page.waitForLoadState('networkidle');
```

## 모달 패턴

### 모던 프레임워크 (React Portal / Vue Teleport)

```javascript
// 모달 열림 대기
const modal = page.getByRole('dialog');
await modal.waitFor({ timeout: 10000 });

// 모달 내 버튼 클릭
await modal.getByRole('button', { name: '확인' }).click();
await modal.getByRole('button', { name: '취소' }).click();

// 모달 닫힘 대기
await modal.waitFor({ state: 'hidden', timeout: 5000 });
```

### 레거시 프레임워크 (iframe 모달)

```javascript
// iframe 내부 접근
const frame = page.frameLocator('iframe[src*="target"]');
await frame.locator('a', { hasText: '실행' }).waitFor({ timeout: 10000 });
await frame.locator('a', { hasText: '실행' }).click();
```

## 테스트 데이터 식별자 패턴

실 운영 데이터와 E2E 테스트 데이터를 목록에서 즉시 구분하기 위한 이름 생성.

```javascript
const KR_CHARS = '김이박최정강조윤장임한오서신권황안송류전홍고문양손배조백허유남심노정하곽성차주우구신임나전민유류진방엄채원천방공강현함변염양변여추도소신석기방편권';
const e2eRandName = () => {
    const c = (arr) => arr[Math.floor(Math.random() * arr.length)];
    const chars = Array.from(KR_CHARS);
    return c(chars) + c(chars) + String(Math.floor(Math.random() * 9000) + 1000);
};
const TEST_NAME = e2eRandName(); // 예: '홍길1234', '정서5154'
```

결과 형태: `홍길1234`, `정서5154` — 한글 성+이름 2자 + 숫자 4자.

## localStorage 기반 파라미터 복원 패턴

PHP 서버 기본값보다 클라이언트 JS의 `localStorage.getItem()`이 나중에 실행되어 우선 적용됨.
파라미터 복원 시 **setItem 먼저 → goto** 순서 필수.

```javascript
// ✅ 올바른 순서 — setItem 먼저, 그 다음 goto
await page.evaluate((val) => {
    try { localStorage.setItem('param_key', String(val)); } catch(_) {}
}, targetValue);
await page.goto(targetUrl, { waitUntil: 'domcontentloaded' });
// goto 후 JS가 localStorage를 읽어 올바른 값 적용

// ❌ 잘못된 순서 — goto 후 setItem은 이번 페이지 로드에는 미반영
await page.goto(targetUrl);
await page.evaluate((val) => localStorage.setItem('key', val), targetValue);
```
